96 template <typename T>
97 bool Testdriver::test_min_heap_correctness(const Queue<T>& queue, int start_index) const {
98  int parent_index = Queue<T>::index_of_parent(start_index);
99  int left_child_index = Queue<T>::index_of_left_child(start_index);
100  int right_child_index = Queue<T>::index_of_right_child(start_index);
101  bool heap_correct = true;
102  bool left_child_exists = queue.get_next() > left_child_index;
103  bool right_child_exists = queue.get_next() > right_child_index;
104  bool parent_exists = parent_index != -1;
105  if (left_child_exists) {
106  heap_correct = test_min_heap_correctness(queue, left_child_index);
107  }
108  if (heap_correct && right_child_exists) {
109  heap_correct = test_min_heap_correctness(queue, right_child_index);
110  }
111  if (parent_exists && heap_correct &&
112  queue.get_entry_at(start_index).get_priority() < queue.get_entry_at(parent_index).get_priority()) {
113  heap_correct = false;
114  }
115  return heap_correct;
116 }
