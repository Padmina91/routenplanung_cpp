177 template <typename T>
178 int Queue<T>::search_value(T val, int start_index) {
179  int found_index = -1;
180  int left_child_index = index_of_left_child(start_index);
181  int right_child_index = index_of_right_child(start_index);
182  bool left_child_exists = _next > left_child_index;
183  bool right_child_exists = _next > right_child_index;
184  T current_val = _entries[start_index].get_value();
185  if (val == current_val) {
186  return start_index;
187  }
188  if (left_child_exists) {
189  found_index = search_value(val, left_child_index);
190  }
191  if (found_index < 0 && right_child_exists) {
192  found_index = search_value(val, right_child_index);
193  }
194  return found_index;

228 template <typename T>

230  int left_child_index = index_of_left_child(start_index);

232  bool left_child_exists = _next > left_child_index;
